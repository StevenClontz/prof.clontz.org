\question Which of these differential equations gives Newton's second law
relating force \(F\), mass \(m\), and position \(x\) (where derivatives are
taken with respect to time)? 
\begin{choices}
\choice \(F'=m/x\) 
\choice \(F''=m/x\) 
\choice \(F/m=x'\) 
\CorrectChoice \(F/m=x''\) %correct
\end{choices}

\question Find the roots of \(2x^2-5x-3\). 
\begin{choices}
\choice \(\frac{1}{3},4\)
\choice None of these. 
\CorrectChoice \(-\frac{1}{2},3\) %correct
\choice \(\frac{1}{4},-1 \)
\end{choices}

\question Simplify \(e^{1+\frac{\pi}{3}i}\).
\begin{choices}
\CorrectChoice \(\frac{e}{2}+\frac{e\sqrt{3}}{2}i\) %correct
\choice \(\frac{e}{3}-\frac{e\sqrt{2}}{2}i\)
\choice None of these. 
\choice \(-\frac{e}{4}+\frac{e}{2}i\)
\end{choices}

\question Find \(\int xe^{2x}\,dx\).
\begin{choices}
\choice None of these. 
\CorrectChoice \(\frac{1}{2}xe^{2x}-\frac{1}{4}e^{2x}+C\) %correct
\choice \(\frac{1}{4}xe^{2x}+\frac{1}{2}e^{2x}+C\)
\choice \(-\frac{1}{8}xe^{2x}+\frac{1}{2}e^{2x}+C\)
\end{choices}

\question Find a solution to the system \(2x-y=3\) and \(3x+2y=8\).
\begin{choices}
\CorrectChoice \((x,y)=(2,1)\) %correct
\choice \((x,y)=(-1,1)\)
\choice \((x,y)=(3,-2)\)
\choice None of these. 
\end{choices}

\end{questions}
